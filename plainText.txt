bookboxd/
├── index.html                   # A "casca" principal do site
├── css/
│   └── style.css                # Estilos visuais e animações (hover das capas)
└── js/
    ├── app.js                   # O Maestro: controla quem entra e o que é desenhado na tela
    │
    ├── services/                # O "Cérebro" de Dados (Banco de Dados Local)
    │   ├── StorageService.js    # Arquivo principal que liga todos os outros serviços
    │   ├── AuthService.js       # Controla logins, senhas e fotos de perfil
    │   ├── ClubService.js       # Controla os livros e o progresso oficial do Clube
    │   └── ExternalService.js   # Controla os livros avulsos adicionados pelos membros
    │
    ├── utils/                   # Ferramentas Lógicas
    │   └── ReadingController.js # A matemática (calcula páginas/capítulos por dia/semana)
    │
    └── components/              # As Telas e Interfaces (O que o usuário vê)
        ├── Login.js             # Tela inicial de entrada
        ├── Dashboard.js         # Menu superior e a aba "Início" (Leitura do Mês)
        ├── Biblioteca.js        # A aba "Biblioteca Central" (Grid com todos os livros)
        └── Gerenciamento.js     # A aba "Gestão" (Editar perfil e área restrita do Admin)
